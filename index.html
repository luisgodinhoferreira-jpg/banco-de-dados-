const canvas = document.getElementById("gameCanvas"); const ctx = canvas.getContext("2d"); const scoreElement = document.getElementById("scoreVal"); // Configurações do Jogador const carWidth = 40; const carHeight = 70; let carX = canvas.width / 2 - carWidth / 2; let carY = canvas.height - 100; let carSpeed = 7; // Obstáculos (Carros Inimigos) let obstacles = []; let obstacleSpeed = 4; let frameCount = 0; let score = 0; let gameOver = false; // Controles let rightPressed = false; let leftPressed = false; document.addEventListener("keydown", (e) => { if (e.key === "ArrowRight") rightPressed = true; if (e.key === "ArrowLeft") leftPressed = true; if (e.key === "Enter" && gameOver) resetGame(); // Reinicia com Enter }); document.addEventListener("keyup", (e) => { if (e.key === "ArrowRight") rightPressed = false; if (e.key === "ArrowLeft") leftPressed = false; }); function drawPlayer() { // Corpo do carro do jogador (Neon Verde) ctx.fillStyle = "#0fa"; ctx.shadowBlur = 15; ctx.shadowColor = "#0fa"; ctx.fillRect(carX, carY, carWidth, carHeight); // Detalhes (Rodas) ctx.fillStyle = "#fff"; ctx.shadowBlur = 0; ctx.fillRect(carX - 5, carY + 10, 5, 15); // Roda esq sup ctx.fillRect(carX + carWidth, carY + 10, 5, 15); // Roda dir sup ctx.fillRect(carX - 5, carY + 45, 5, 15); // Roda esq inf ctx.fillRect(carX + carWidth, carY + 45, 5, 15); // Roda dir inf } function drawObstacles() { obstacles.forEach((obj, index) => { // Estilo do carro inimigo (Neon Rosa/Roxo) ctx.fillStyle = "#f0f"; ctx.shadowBlur = 15; ctx.shadowColor = "#f0f"; ctx.fillRect(obj.x, obj.y, carWidth, carHeight); obj.y += obstacleSpeed; // Detecção de Colisão if ( carX < obj.x + carWidth && carX + carWidth > obj.x && carY < obj.y + carHeight && carY + carHeight > obj.y ) { gameOver = true; } // Remove carros que saíram da tela e soma pontos if (obj.y > canvas.height) { obstacles.splice(index, 1); score++; scoreElement.innerText = score; // Aumenta a dificuldade gradualmente if (score % 5 === 0) obstacleSpeed += 0.2; } }); } function update() { if (gameOver) { ctx.fillStyle = "white"; ctx.font = "30px Segoe UI"; ctx.textAlign = "center"; ctx.fillText("FIM DE JOGO!", canvas.width / 2, canvas.height / 2); ctx.font = "20px Segoe UI"; ctx.fillText("Pressione ENTER para reiniciar", canvas.width / 2, canvas.height / 2 + 40); return; } ctx.clearRect(0, 0, canvas.width, canvas.height); // Movimentação do jogador if (rightPressed && carX < canvas.width - carWidth) carX += carSpeed; if (leftPressed && carX > 0) carX -= carSpeed; // Criar novos carros a cada 80 frames frameCount++; if (frameCount % 80 === 0) { let laneX = Math.random() * (canvas.width - carWidth); obstacles.push({ x: laneX, y: -carHeight }); } drawObstacles(); drawPlayer(); requestAnimationFrame(update); } function resetGame() { carX = canvas.width / 2 - carWidth / 2; obstacles = []; score = 0; obstacleSpeed = 4; scoreElement.innerText = score; gameOver = false; frameCount = 0; update(); } update();
